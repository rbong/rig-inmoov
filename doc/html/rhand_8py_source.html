68 def sweep (initial, servos, starts, ends, steps):
69  """! Produces complex command chains.
70 
71  The output returned from the function is formatted in such a way that the
72  commands can be transmitted to the serial port at variable speeds.
73 
74  The parameters must all be lists of the same length whose indexes
75  correspond with \b servos.
76 
77  @param initial An array of initial commands to append to.
78  @param servos The servos to control.
79  @param starts The current positions of the servos.
80  @param ends The desired end positions.
81  @param steps The values to use to decrement/increment.
82  Converted to absolute value.
83 
84  @returns Upon success, returns the generated list of commands.
85  The commands are surrounded by the @ref CANCEL_SIGNAL for syncing.
86  The servos will always reach their exact final destination, unless if
87  \b steps at the index of the servo is 0.
88  Upon error, raises <b>'Invalid sweep command'</b>.
89  This indicates non-matching list lengths.
90 
91  @par Example
92  @code sweep ([], [1,2], [180,0], [90,180], [90,45]) @endcode
93  Returns
94  @code
95  [CANCEL_SIGNAL, 1, 180, 2, 0, 1, 90, 2, 45, 2, 90, 2, 135, 2, 180, CANCEL_SIGNAL]
96  @endcode
97  The protocol this follows is described in servo.ino.
98  By intertwining commands in this way, a delay can be inserted between
99  bytes and fluid motion is still preserved.
100  """
101 
102  # current angles of the servos
103  pos = list (starts)
104 
105  # length of the servo list
106  ln = len (servos)
107 
108  if ln != len (starts) or ln != len (ends) or ln != len (steps):
109  raise NameError ('Invalid sweep command')
110 
111  initial.append (CANCEL_SIGNAL)
112 
113  # the done list indicates which servos have completed their gestures
114  done = [0] * ln
115  while 1:
116  i = 0
117  while i < ln:
118  if done [i]:
119  i += 1
120  continue
121 
122  print ('servo:')
123  print (servos [i])
124  print ('pos:')
125  print (pos [i])
126  initial.append (servos [i])
127  initial.append (pos [i])
128 
129  # servo has reached its final position, indicate completion
130  if pos [i] == ends [i] or steps [i] == 0:
131  done [i] = 1
132  # the starting position preceeds the ending position, increment
133  elif starts [i] < ends [i]:
134  pos [i] += abs (steps [i])
135  # ensure that the servo reaches its exact destination
136  if pos [i] > ends [i]:
137  pos [i] = ends [i]
138  # the ending position preceeds the starting position, decrement
139  else:
140  pos [i] -= abs (steps [i])
141  # ensure that the servo reaches its exact destination
142  if pos [i] < ends [i]:
143  pos [i] = ends [i]
144  i += 1
145 
146  if not (0 in done):
147  break
148 
149  initial.append (CANCEL_SIGNAL)
150  return initial

204 def dump (servos=6):
205  """! Retrieves information about the board.
206  The command sends the @ref DUMP_SIGNAL to the serial port and reads back
207  the response. The response is described in servo.ino.
208 
209  When you write your own similar function, timing is important.
210 
211  We may need to increase the response delay on the Arduino if you are unable
212  to recieve a response. View the source of this function to understand how
213  it works, but keep in mind it is untested. Also keep in mind that the
214  function is very non-general as it always reads 10 bytes instead of looking
215  for the end signal.
216 
217  @note While this function could be used to change dynamically between
218  gestures instead of resetting the servos to 0, this function did not exist
219  when this script was first written.
220  """
221 
222  # ensure there is no pending serial input
223  ser.flushInput ()
224 
225  # indicate to the board that we with to retrieve information
226  servowrite ([DUMP_SIGNAL], 0)
227 
228  # read 10 bytes- the begin byte, the id of the board, the number of servos,
229  # the positions of the servos (6 servos on the hand), and the end byte
230  response = ser.read (servos + 4)
231  # convert the list of bytes into a list of ints
232  response = list (response)
233 
234  # terminate if the start or end byte is not correct
235  if not (response [0] == DUMP_START_RESPONSE and response [-1] == DUMP_END_RESPONSE):
236  return
237 
238  print ('id: ' + str (response [1]))
239  if response [1] == RHAND_ID:
240  print ('this board is the right hand.')
241  print ('servos: ' + str (response [2]))
242  servos = response [2]
243  i = 0
244  while i < servos:
245  print ('servo ' + str (i) + ': ' + str (response [i + 3]))
246  i += 1
