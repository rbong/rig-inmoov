214 void setAdjustedAngles (uint8_t servo_num)
215 {
216  // doubles used for conversion
217  double dangle, dmin, dmax;
218 
219  for (uint8_t angle = 0; angle <= 180; angle++)
220  {
221  // arduino has its own map function, but it's broken on some systems
222  // conversion to double avoids integer rounding errors
223  dangle = angle;
224  dmin = limit [servo_num] [MIN_LIM];
225  dmax = limit [servo_num] [MAX_LIM];
226  dangle = dmin + ((dangle * (dmax - dmin)) / 180);
227  adjusted_angles [servo_num] [angle] = dangle;
228  }
