101 def sweep (initial, servos, starts, ends, steps):
102  """! Produces complex command chains.
103 
104  The output returned from the function is formatted in such a way that the
105  commands can be transmitted to the serial port at variable speeds.
106 
107  The parameters must all be lists of the same length whose indexes
108  correspond with \b servos.
109 
110  @param initial An array of initial commands to append to.
111  @param servos The servos to control.
112  @param starts The current positions of the servos.
113  @param ends The desired end positions.
114  @param steps The values to use to decrement/increment.
115  Converted to absolute value.
116 
117  @returns Upon success, returns the generated list of commands.
118  The commands are surrounded by the @ref CANCEL_SIGNAL for syncing.
119  The servos will always reach their exact final destination, unless if
120  \b steps at the index of the servo is 0.
121  Upon error, raises <b>'Invalid sweep command'</b>.
122  This indicates non-matching list lengths.
123 
124  @par Example
125  @code sweep ([], [1,2], [180,0], [90,180], [90,45]) @endcode
126  Returns
127  @code
128  [CANCEL_SIGNAL, 1, 180, 2, 0, 1, 90, 2, 45, 2, 90, 2, 135, 2, 180, CANCEL_SIGNAL]
129  @endcode
130  The protocol this follows is described in servo.ino.
131  By intertwining commands in this way, a delay can be inserted between
132  bytes and fluid motion is still preserved.
133  """
134 
135  # current angles of the servos
136  pos = list (starts)
137 
138  # length of the servo list
139  ln = len (servos)
140 
141  if ln != len (starts) or ln != len (ends) or ln != len (steps):
142  raise NameError ('Invalid sweep command')
143 
144  initial.append (CANCEL_SIGNAL)
145 
146  # the done list indicates which servos have completed their gestures
147  done = [0] * ln
148  while 1:
149  i = 0
150  while i < ln:
151  if done [i]:
152  i += 1
153  continue
154 
155  print ('servo:')
156  print (servos [i])
157  print ('pos:')
158  print (pos [i])
159  initial.append (servos [i])
160  initial.append (pos [i])
161 
162  # servo has reached its final position, indicate completion
163  if pos [i] == ends [i] or steps [i] == 0:
164  done [i] = 1
165  # the starting position preceeds the ending position, increment
166  elif starts [i] < ends [i]:
167  pos [i] += abs (steps [i])
168  # ensure that the servo reaches its exact destination
169  if pos [i] > ends [i]:
170  pos [i] = ends [i]
171  # the ending position preceeds the starting position, decrement
172  else:
173  pos [i] -= abs (steps [i])
174  # ensure that the servo reaches its exact destination
175  if pos [i] < ends [i]:
176  pos [i] = ends [i]
177  i += 1
178 
179  if not (0 in done):
180  break
181 
182  initial.append (CANCEL_SIGNAL)
183  return initial

239 def dump (servos=6):
240  """! Retrieves information about the board.
241  The command sends the @ref DUMP_SIGNAL to the serial port and reads back
242  the response. The response is described in servo.ino.
243 
244  When you write your own similar function, timing is important.
245 
246  We may need to increase the response delay on the Arduino if you are unable
247  to recieve a response. View the source of this function to understand how
248  it works, but keep in mind it is untested. Also keep in mind that the
249  function is very non-general as it always reads the same amount of bytes
250  instead of looking for the end signal.
251 
252  @note While this function could be used to change dynamically between
253  gestures instead of resetting the servos to 0, this function did not exist
254  when this script was first written.
255  """
256 
257  # ensure there is no pending serial input
258  ser.flushInput ()
259 
260  # indicate to the board that we with to retrieve information
261  servowrite ([DUMP_SIGNAL], 0)
262 
263  # read 10 bytes- the begin byte, the id of the board, the number of servos,
264  # the positions of the servos (6 servos on the hand), and the end byte
265  response = ser.read (servos * 2 + 4)
266  # convert the list of bytes into a list of ints
267  response = list (response)
268 
269  # terminate if the start or end byte is not correct
270  if not (response [0] == START_RESPONSE and response [-1] == END_RESPONSE):
271  return
272 
273  print ('id: ' + str (response [1]))
274  if response [1] == RHAND_ID:
275  print ('this board is the right hand.')
276  print ('servos: ' + str (response [2]))
277  servos = response [2]
278  i = 0
279  while i < servos * 2:
280  print ('servo ' + str (response [i + 3]) + ': ' + str (response [i + 4]))
281  i += 2

341 def getarrow ():
342  import termios
343  import sys, tty
344  def _getch():
345  fd = sys.stdin.fileno()
346  old_settings = termios.tcgetattr(fd)
347  try:
348  tty.setraw(fd)
349  ch = sys.stdin.read(1)
350  finally:
351  termios.tcsetattr(fd, termios.TCSADRAIN, old_settings)
352  return ch
353 
354  c = _getch ()
355  if c != '\x1b':
356  return ''
357  c += _getch ()
358  if c != '\x1b[':
359  return ''
360  c += _getch ()
361 
362  if c == '\x1b[A':
363  return 'up'
364  elif c == '\x1b[B':
365  return 'down'
366  elif c == '\x1b[C':
367  return 'right'
368  elif c == '\x1b[D':
369  return 'left'
370  else:
371  return ''

373 def calibrate (s=1):
374  """! Calibrate a servo.
375  Takes one argument, which is the id of a server. @ref index has the servo
376  codes.
377 
378  Use the left and right arrow keys to adjust the value of the servo. Use any
379  other key to exit.
380 
381  Enter the maximum values into the @ref limit array in your settings file.
382 
383  Note that if a servo isn't moving the correct way, for example if a finger
384  isn't flat at 0 and curved at 180, you must set @ref reverse to 1 at the
385  servo index.
386  """
387  i = 90
388  while 1:
389  print (i)
390  ser.write ([255, s, i])
391  arrow = getarrow ()
392  if arrow == 'up' or arrow == 'right':
393  if i < 180:
394  i += 5
395  elif arrow == 'down' or arrow == 'left':
396  if i > 0:
397  i -= 5
398  else:
399  break
