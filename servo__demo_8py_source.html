92 def sweep (initial, servos, starts, ends, steps):
93  """! Produces complex command chains.
94 
95  The output returned from the function is formatted in such a way that the
96  commands can be transmitted to the serial port at variable speeds.
97 
98  The parameters must all be lists of the same length whose indexes
99  correspond with \b servos.
100 
101  @param initial An array of initial commands to append to.
102  @param servos The servos to control.
103  @param starts The current positions of the servos.
104  @param ends The desired end positions.
105  @param steps The values to use to decrement/increment.
106  Converted to absolute value.
107 
108  @returns Upon success, returns the generated list of commands.
109  The commands are surrounded by the @ref CANCEL_SIGNAL for syncing.
110  The servos will always reach their exact final destination, unless if
111  \b steps at the index of the servo is 0.
112  Upon error, raises <b>'Invalid sweep command'</b>.
113  This indicates non-matching list lengths.
114 
115  @par Example
116  @code sweep ([], [1,2], [180,0], [90,180], [90,45]) @endcode
117  Returns
118  @code
119  [CANCEL_SIGNAL, 1, 180, 2, 0, 1, 90, 2, 45, 2, 90, 2, 135, 2, 180, CANCEL_SIGNAL]
120  @endcode
121  The protocol this follows is described in servo.ino.
122  By intertwining commands in this way, a delay can be inserted between
123  bytes and fluid motion is still preserved.
124  """
125 
126  # current angles of the servos
127  pos = list (starts)
128 
129  # length of the servo list
130  ln = len (servos)
131 
132  if ln != len (starts) or ln != len (ends) or ln != len (steps):
133  raise NameError ('Invalid sweep command')
134 
135  initial.append (CANCEL_SIGNAL)
136 
137  # the done list indicates which servos have completed their gestures
138  done = [0] * ln
139  while 1:
140  i = 0
141  while i < ln:
142  if done [i]:
143  i += 1
144  continue
145 
146  print ('servo:')
147  print (servos [i])
148  print ('pos:')
149  print (pos [i])
150  initial.append (servos [i])
151  initial.append (pos [i])
152 
153  # servo has reached its final position, indicate completion
154  if pos [i] == ends [i] or steps [i] == 0:
155  done [i] = 1
156  # the starting position preceeds the ending position, increment
157  elif starts [i] < ends [i]:
158  pos [i] += abs (steps [i])
159  # ensure that the servo reaches its exact destination
160  if pos [i] > ends [i]:
161  pos [i] = ends [i]
162  # the ending position preceeds the starting position, decrement
163  else:
164  pos [i] -= abs (steps [i])
165  # ensure that the servo reaches its exact destination
166  if pos [i] < ends [i]:
167  pos [i] = ends [i]
168  i += 1
169 
170  if not (0 in done):
171  break
172 
173  initial.append (CANCEL_SIGNAL)
174  return initial

230 def dump (servos=6):
231  """! Retrieves information about the board.
232  The command sends the @ref DUMP_SIGNAL to the serial port and reads back
233  the response. The response is described in servo.ino.
234 
235  When you write your own similar function, timing is important.
236 
237  We may need to increase the response delay on the Arduino if you are unable
238  to recieve a response. View the source of this function to understand how
239  it works, but keep in mind it is untested. Also keep in mind that the
240  function is very non-general as it always reads the same amount of bytes
241  instead of looking for the end signal.
242 
243  @note While this function could be used to change dynamically between
244  gestures instead of resetting the servos to 0, this function did not exist
245  when this script was first written.
246  """
247 
248  # ensure there is no pending serial input
249  ser.flushInput ()
250 
251  # indicate to the board that we with to retrieve information
252  servowrite ([DUMP_SIGNAL], 0)
253 
254  # read 10 bytes- the begin byte, the id of the board, the number of servos,
255  # the positions of the servos (6 servos on the hand), and the end byte
256  response = ser.read (servos * 2 + 4)
257  # convert the list of bytes into a list of ints
258  response = list (response)
259 
260  # terminate if the start or end byte is not correct
261  if not (response [0] == START_RESPONSE and response [-1] == END_RESPONSE):
262  return
263 
264  print ('id: ' + str (response [1]))
265  if response [1] == RHAND_ID:
266  print ('this board is the right hand.')
267  print ('servos: ' + str (response [2]))
268  servos = response [2]
269  i = 0
270  while i < servos * 2:
271  print ('servo ' + str (response [i + 3]) + ': ' + str (response [i + 4]))
272  i += 2

332 def getarrow ():
333  import termios
334  import sys, tty
335  def _getch():
336  fd = sys.stdin.fileno()
337  old_settings = termios.tcgetattr(fd)
338  try:
339  tty.setraw(fd)
340  ch = sys.stdin.read(1)
341  finally:
342  termios.tcsetattr(fd, termios.TCSADRAIN, old_settings)
343  return ch
344 
345  c = _getch ()
346  if c != '\x1b':
347  return ''
348  c += _getch ()
349  if c != '\x1b[':
350  return ''
351  c += _getch ()
352 
353  if c == '\x1b[A':
354  return 'up'
355  elif c == '\x1b[B':
356  return 'down'
357  elif c == '\x1b[C':
358  return 'right'
359  elif c == '\x1b[D':
360  return 'left'
361  else:
362  return ''

364 def calibrate (s=1):
365  """! Calibrate a servo.
366  Takes one argument, which is the id of a server. @ref index has the servo
367  codes.
368 
369  Use the left and right arrow keys to adjust the value of the servo. Use any
370  other key to exit.
371 
372  Enter the maximum values into the @ref limit array in your settings file.
373 
374  Note that if a servo isn't moving the correct way, for example if a finger
375  isn't flat at 0 and curved at 180, you must set @ref reverse to 1 at the
376  servo index.
377  """

379  while 1:
380  print (i)

383  if arrow == 'up' or arrow == 'right':
384  if i < 180:
385  i += 5
386  elif arrow == 'down' or arrow == 'left':
387  if i > 0:
388  i -= 5
